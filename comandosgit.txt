ls: listar carpetas
pwd: saber donde estamos
cd: entrar o salir de una carpeta
mkdir: crear carpeta
ls -a: muestra todos los archivos y carpetas
code .: abrir carpeta donde estamos
git status: muestra el estado actual del repositorio
git add: seleccionar archivos
git commit -m "comentario": Archivos comprometidos 
rm: eliminar archivos
git rm: se pasa altiro para poder ser comiteado
git restore: recuperar archivo
mv: mover o renombrar archivos
git mv: se pasa altiro para poder ser comiteado
git status -s: estado más sofisticado
git diff: muestra todos los cambios escritos que aun no han pasado a staged
git diff --staged: mostrara todos los cambios ya agregados a staged
git log: historial 
git log --oneline: historial más reducido
git branch: indica la rama donde estamos trabajando
git checkout -b: crear una rama
cat: muestra el contenido del archivo
git checkout: cambiar de rama
